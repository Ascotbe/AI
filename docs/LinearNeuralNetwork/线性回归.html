def get_data(num, w=2.0, b=3.0):
    for _ in range(num):
        x = np.random.uniform(-10.0, 10.0)
        noise = np.random.normal(0, 1)

        yield np.array([x]).astype(np.float32), np.array([y]).astype(np.float32)
